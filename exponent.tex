\textbf{【指数の拡張】}\\
\textbf{指数法則}
\begin{itemize}
        \item \(a^m \times a^n = \textcolor[rgb]{1,0.5,0.5}{a^{m + n}}\)
        \item \(\frac{a^m}{a^n} = \textcolor[rgb]{1,0.5,0.5}{a^{m - n}}\)
        \item \( (a^m)^n = \textcolor[rgb]{1,0.5,0.5}{a^{mn}}\)
        \item \( (ab)^n = \textcolor[rgb]{1,0.5,0.5}{a^nb^n}\)
\end{itemize}
